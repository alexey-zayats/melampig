	ct.sorder as "жеребьевка", 
	t.title as "команда", 
	ct.classify_points as "классификационные очки",

	competition_team as ct, 
	team as t ,
	cgroup as g
where 
	ct.competition = 1308 
	and t.id = ct.team 
	and g.id = ct.cgroup

	3 desc,
	4 desc,
	5,
	6 desc,
	7
